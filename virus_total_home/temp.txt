import os
import glob


def file_check(target_name):
    """檢查檔案是否存在，三種檔案的內容排列必需一列一筆"""
    result = glob.glob(f'{os.getcwd()}/{target_name}.*')  # 抓出此檔案的絕對路徑，加上特定檔名
    check_list = list()  # 過濾完的結果
    file_extension = ['txt', 'csv', 'xls']  # 特定副檔名
    for i in result:  # 迭代找到指定檔名的檔案
        for k in file_extension:  # 濾出特定副檔名
            if i.endswith(k):
                check_list.append(i)
    return check_list

import os
import time
import csv
import xlrd
import catch_file
import all_func


class Operation:

    def __init__(self, receive_func):
        self.receive_func = receive_func

    def file_check(self):
        check_result = catch_file.file_check(self.receive_func)
        if len(check_result) == 0:
            return f"沒找到檔案名稱為{self.receive_func}的檔案"
        for judgment_extension in check_result:
            extension = os.path.splitext(judgment_extension)[-1]  # 檢查副檔名，呼叫不同的讀取文件方式
            if extension in '.csv':
                Operation(self).excel_csv(judgment_extension)
            elif extension in '.txt':
                Operation(self).read_txt(judgment_extension)
            else:
                Operation(self).excel_xls(judgment_extension)

    def judgment_func(self, file_name, data):
        if 'domaindata' in file_name:
            all_func.General().domain(data)
        elif 'urldata' in file_name:
            all_func.General().urls(data)
        elif 'filedata' in file_name:
            all_func.General().file(data)
        else:
            all_func.General().ip(data)

    def excel_csv(self, file):
        """讀取csv方法"""
        open_file = open(file, 'r', newline='', encoding='utf-8-sig')
        rows = csv.reader(open_file, delimiter=',')
        for top_shelf in rows:  # 從csv逐行讀出是list
            for second_shelf in top_shelf:  # 需再一次loop輸出str
                Operation(self).judgment_func(file, second_shelf)
                time.sleep(15)  # 免費帳號的等待時間

    def excel_xls(self, file):
        """讀取xls方法(可讀取多分頁)"""
        wb = xlrd.open_workbook(file)  # 讀取指定xls檔
        sheets = wb.sheets()  # 掃出該檔所有sheet
        for sheet in sheets:
            rows = sheet.get_rows()  # 取得所有列
            for row in rows:
                Operation(self).judgment_func(file, row[0].value)
                time.sleep(15)  # 免費帳號的等待時間

    def read_txt(self, file):
        """讀取txt的方法，一列一筆資料後直接換行"""
        openfile = open(file, 'r')
        lines = openfile.readlines()
        for i in range(len(lines)):
            line = lines[i].replace("\n", "")
            Operation(self).judgment_func(file, line)
            time.sleep(15)
